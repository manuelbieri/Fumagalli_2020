113        """
114        Generates a legend for the parameter values of a Fumagalli_Motta_Tarantino_2020.Models.BaseModel in latex format.

131            ("B", model.private_benefit, separator_parameters),
132            ("K", model.development_costs, separator_parameters),
133            ("p", model.success_probability, "\n"),
134            ("CS^m", model.cs_without_innovation, separator_parameters),
135            (
136                "\\pi^m_I",
137                model.incumbent_profit_without_innovation,
138                separator_parameters,
139            ),
140            ("CS^M", model.cs_with_innovation, separator_parameters),
141            ("\\pi^M_I", model.incumbent_profit_with_innovation, separator_parameters),
142            ("CS^d", model.cs_duopoly, separator_parameters),
143            ("\\pi^d_I", model.incumbent_profit_duopoly, separator_parameters),
144            ("\\pi^d_S", model.startup_profit_duopoly, ""),

261class AssetRange(IVisualize):
262    """
263    Visualizes the outcomes over an assets range for a specific model.
264    """
265
266    def __init__(self, model: Models.OptimalMergerPolicy, **kwargs) -> None:
267        super(AssetRange, self).__init__(model, **kwargs)
268        self.labels: list[str] = []
269        self.colors: dict[str, str] = {}
270
271    def _get_outcomes_asset_range(
272        self,
273    ) -> list[Types.OptimalMergerPolicySummary]:
274        """
275        Generates a list with all essential threshold concerning the assets of a start-up and an additional list with
276        summaries of the outcomes of the model in between the thresholds.
277
278        Returns
279        -------
280        (list[Fumagalli_Motta_Tarantino_2020.Types.ThresholdItem], list[Fumagalli_Motta_Tarantino_2020.Types.OptimalMergerPolicySummary])
281            List containing the essential asset thresholds in the model and list containing the summaries of the outcomes of the model.
282        """
283        asset_range: list[Types.ThresholdItem] = self._get_asset_thresholds()
284        summaries: list[Types.OptimalMergerPolicySummary] = []
285        for i in range(len(asset_range) - 1):
286            self.model.startup_assets = (
287                self.model.asset_distribution.inverse_cumulative(asset_range[i].value)
288                + self.model.asset_distribution.inverse_cumulative(
289                    asset_range[i + 1].value
290                )
291            ) / 2
292            summaries.append(self.model.summary())
293        return summaries
294
295    def _get_asset_thresholds(self) -> list[Types.ThresholdItem]:
296        """
297        Generates a list with all essential threshold concerning the assets of a start-up.
298
299        Returns
300        -------
301        list[Fumagalli_Motta_Tarantino_2020.Types.ThresholdItem]
302            List containing the essential asset thresholds in the model.
303        """
304        min_threshold = Types.ThresholdItem("0.5", 0.5)
305        max_threshold = Types.ThresholdItem(
306            "$F(K)$",
307            self.model.asset_distribution.cumulative(self.model.development_costs),
308        )
309        thresholds = self._get_essential_thresholds()
310        essential_thresholds: list[Types.ThresholdItem] = []
311        for threshold in thresholds:
312            if min_threshold.value < threshold.value < max_threshold.value:
313                essential_thresholds.append(threshold)
314        thresholds = sorted(essential_thresholds, key=lambda x: x.value)
315        thresholds.insert(0, min_threshold)
316        thresholds.append(max_threshold)
317        return thresholds
318
319    def _get_essential_thresholds(self):
320        thresholds: list[Types.ThresholdItem] = [
321            Types.ThresholdItem(
322                "$\\Gamma$", self.model.asset_distribution_threshold_strict
323            ),
324            Types.ThresholdItem("$\\Phi$", self.model.asset_distribution_threshold),
325            Types.ThresholdItem(
326                "$\\Phi^T$", self.model.asset_distribution_threshold_laissez_faire
327            ),
328            Types.ThresholdItem(
329                "$\\Phi^{\\prime}$",
330                self.model.asset_distribution_threshold_intermediate,
331            ),
332            Types.ThresholdItem("$F(\\bar{A})$", self.model.asset_threshold_cdf),
333            Types.ThresholdItem(
334                "$F(\\bar{A}^T)$", self.model.asset_threshold_late_takeover_cdf
335            ),
336        ]
337        return thresholds

339    @staticmethod
340    def _get_x_labels_ticks(
341        asset_thresholds: list[Types.ThresholdItem],

343        """
344        Generates the locations of the ticks on the x-axis and the corresponding labels on the x-axis.
345
346        Parameters

363    def _set_x_ticks(self, asset_thresholds: list[Types.ThresholdItem]) -> None:
364        x_ticks, x_labels = self._get_x_labels_ticks(asset_thresholds)
365        self.ax.xaxis.set_major_locator(FixedLocator(x_ticks[::2]))
366        self.ax.xaxis.set_minor_locator(FixedLocator(x_ticks[1::2]))
367        self.ax.set_xticklabels(x_labels[::2])
368        self.ax.set_xticklabels(x_labels[1::2], minor=True)
369        self.ax.tick_params(
370            which="minor",
371            bottom=False,
372            top=True,
373            labelbottom=False,
374            labeltop=True,
375            axis="x",
376        )
377        self.ax.tick_params(which="both", length=6, axis="x")
378        for threshold in asset_thresholds:
379            if threshold.value != max(item.value for item in asset_thresholds):
380                self.ax.axvline(threshold.value, linestyle="--", color="k")

393    def _get_label_color(self, label) -> (str, str):
394        """
395        Returns the color and the final label for a legend entry.
396
397        Through this method, duplications in the legend are avoided.
398
399        Parameters
400        ----------
401        label: str
402
403        Returns
404        -------
405        (str, str)
406            String representing the final label and a string representing the color.
407        """
408        if label in self.labels:
409            return "_nolegend_", self.colors[label]
410        self.colors[label] = IVisualize.colors[len(self.labels)]
411        self.labels.append(label)
412        return label, self.colors[label]
413
414    def _get_summaries(self) -> list[list[Types.OptimalMergerPolicySummary]]:
415        return [self._get_outcomes_asset_range()]
416
417    def plot(self, **kwargs) -> (plt.Figure, plt.Axes):
418        asset_range = self._get_asset_thresholds()
419        merger_policies_summaries = self._get_summaries()
420        assert asset_range is not None
421        assert merger_policies_summaries is not None
422        self.labels.clear()
423        self.colors.clear()
424
425        spacing: float = kwargs.get("spacing", 0.1)
426        bar_height: float = kwargs.get("bar_height", 0.2)
427        y_labels: list[str] = []
428        for number_merger_policy, summaries in enumerate(merger_policies_summaries):
429            y_labels.append(summaries[0].set_policy.abbreviation())
430            for summary_index, summary in enumerate(summaries):
431                length: float = (
432                    asset_range[summary_index + 1].value
433                    - asset_range[summary_index].value
434                )
435                label: str = self._get_summary_latex(summary)
436                label, color = self._get_label_color(label)
437                self.ax.barh(
438                    y=spacing * (number_merger_policy + 1)
439                    + bar_height * number_merger_policy,
440                    width=length,
441                    left=asset_range[summary_index].value,
442                    height=bar_height,
443                    color=color,
444                    label=label,
445                )
446        self._set_legend()
447        if kwargs.get("legend", True):
448            self.ax.annotate(
449                self._get_symbol_legend(),
450                xy=(asset_range[0].value, 0),
451                xytext=(0, -50),
452                textcoords="offset points",
453                horizontalalignment="left",
454                verticalalignment="top",
455            )
456        self.ax.margins(y=spacing, x=0)
457        self._set_x_ticks(asset_range)
458        self._set_y_ticks(bar_height, spacing, y_labels)
459        self.ax.set_xlabel("Cumulative Distribution Value of Assets $F(A)$")
460        self.ax.set_ylabel("Merger Policy")
461        self._set_tight_layout()
462        return self.fig, self.ax

631
